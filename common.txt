  def findB[A](v: A,ls: Array[A]):Int = {
    var mid = (1 + ls.length/2).toInt
    for (x <- 0 to mid) {
      if(ls(x) == v) return x;
    }
     for (y <- (mid to ls.length).reverse) {
      if(ls(y) == v) return y;
    }
